def f(x:float, y:float):
    return  x**2 + 3 * y

def dfx(x:float):
    return 2 * x

def dfy(y:float):
    return 3

Implementacion¶

In [ ]:
x_values = []
y_values = []
fxy_values = []
x = INITICIAL_X
y = INITICIAL_Y

for _ in range(NUM_ITERATIONS):
    x_values.append(x)
    y_values.append(y)
    fxy_values.append( f(x, y) )
    x = x - LERNING_RATE * dfx(x)
    y = y - LERNING_RATE * dfy(y)
